1   package dessin.formes;
2   import java.awt.Color;
3   import java.awt.Dimension;
4   import java.awt.Graphics;
5   import java.awt.event.MouseEvent;
6   import java.util.HashSet;
7   import java.util.Set;
8   
9   public class GroupeFormes extends AbstractForme {
10  	private Set<AbstractForme> contenu;
11  	private boolean selectionne;
12  	
13  	public GroupeFormes() {
14  		super();
15  		contenu = new HashSet<>();
16  		selectionne = false;
17  		this.setBounds(0, 0, 600, 300);
18  	}
19  	
20  	public Set<AbstractForme> vider() {
21  		Set<AbstractForme> retour = contenu;
22  		contenu = new HashSet<>();
23  		selectionne = false;
24  		return retour;
25  	}
26  	
27  	public void ajouterForme(AbstractForme f) {
28  		contenu.add(f);
29  		f.setSelectionne(false);
30  	}
31  
32  	@Override
33  	public void setSelectionne(boolean b) {
34  		selectionne = b;
35  		repaint();
36  	}
37  
38  	@Override
39  	public void setRempli(boolean b) {
40  		contenu.stream().forEach(x -> x.setRempli(b));
41  		repaint();
42  	}
43  
44  	@Override
45  	public boolean isRempli() {
46  		return contenu.stream().allMatch(Forme::isRempli);
47  	}
48  
49  	@Override
50  	public void agrandir() {
51  		contenu.stream().forEach(Forme::agrandir);	
52  		repaint();
53  	}
54  
55  	@Override
56  	public void diminuer() {
57  		contenu.stream().forEach(Forme::diminuer);
58  		repaint();
59  	}
60  
61  	@Override
62  	public Dimension getPreferredSize() {
63  		return new Dimension(600,300);
64  	}
65  
66  	@Override
67  	public void paint(Graphics g) {
68  		contenu.stream().forEach(f -> f.paint(g));
69  		if (selectionne) {
70  			System.err.println(String.format("cadre : %d,%d,%d,%d", getXmin(), getXmax(), getYmin(), getYmax()));
71  
72  			g.setColor(Color.LIGHT_GRAY);
73  			g.drawRect(getXmin(), getYmin(), getXmax()-getXmin(), getYmax()-getYmin());
74  		}
75  	}
76  
77  	@Override
78  	public boolean clique(MouseEvent me) {
79  		System.err.println("clic en "+ me.getX() + "," +  me.getY());
80  		System.err.println(String.format("%d,%d,%d,%d", getXmin(), getXmax(), getYmin(), getYmax()));
81  		return me.getX() >= getXmin() &&
82  				me.getX() <= getXmax() &&
83  				me.getY() >= getYmin() &&
84  				me.getY() <= getYmax();
85  		
86  	}
87  
88  	@Override
89  	public int getWidth() {
90  		return 600;
91  	}
92  	@Override
93  	public int getHeight() {
94  		return 300;
95  	}
96  	
97  	@Override
98  	public int getXmin() {
99  		return contenu.stream().mapToInt(Forme::getXmin).min().getAsInt();
100 	}
101 
102 	@Override
103 	public int getXmax() {
104 		return contenu.stream().mapToInt(Forme::getXmax).max().getAsInt();
105 	}
106 
107 	@Override
108 	public int getYmin() {
109 		return contenu.stream().mapToInt(Forme::getYmin).min().getAsInt();
110 	}
111 
112 	@Override
113 	public int getYmax() {
114 		return contenu.stream().mapToInt(Forme::getYmax).max().getAsInt();
115 	}
116 
117 	@Override
118 	public String toString() {
119 		return "GroupeCercle [contenu=" + contenu + "]";
120 	}
121 	
122 	
123 }
